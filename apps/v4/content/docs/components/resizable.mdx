## 用法

```tsx showLineNumbers
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from "@/components/ui/resizable"
```

```tsx showLineNumbers
<ResizablePanelGroup direction="horizontal">
  <ResizablePanel>面板一</ResizablePanel>
  <ResizableHandle />
  <ResizablePanel>面板二</ResizablePanel>
</ResizablePanelGroup>
```

## Examples

### Vertical

通过 `direction` 属性控制面板的伸缩方向。

<ComponentPreview
  name="resizable-vertical"
  description="A group of resizable vertical panels."
/>

```tsx showLineNumbers {9}
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from "@/components/ui/resizable"

export default function Example() {
  return (
    <ResizablePanelGroup direction="vertical">
      <ResizablePanel>面板一</ResizablePanel>
      <ResizableHandle />
      <ResizablePanel>面板二</ResizablePanel>
    </ResizablePanelGroup>
  )
}
```

### Handle

可通过 `ResizableHandle` 的 `withHandle` 属性设置或隐藏拖拽手柄。

<ComponentPreview
  name="resizable-handle"
  description="A group of resizable panels with a handle."
/>

```tsx showLineNumbers {11}
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from "@/components/ui/resizable"

export default function Example() {
  return (
    <ResizablePanelGroup direction="horizontal">
      <ResizablePanel>面板一</ResizablePanel>
      <ResizableHandle withHandle />
      <ResizablePanel>面板二</ResizablePanel>
    </ResizablePanelGroup>
  )
}
```
