### 链接

可以通过 `asChild` 属性让其他组件呈现徽章样式。下面展示一个看起来像徽章的链接。

```tsx showLineNumbers
import Link from "next/link"

import { Badge } from "@/components/ui/badge"

export function LinkAsBadge() {
  return (
    <Badge asChild>
      <Link href="/">Badge</Link>
    </Badge>
  )
}
```
