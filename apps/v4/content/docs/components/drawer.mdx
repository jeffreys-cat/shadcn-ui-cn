## 用法

```tsx showLineNumbers
import {
  Drawer,
  DrawerClose,
  DrawerContent,
  DrawerDescription,
  DrawerFooter,
  DrawerHeader,
  DrawerTitle,
  DrawerTrigger,
} from "@/components/ui/drawer"
```

```tsx showLineNumbers
<Drawer>
  <DrawerTrigger>打开</DrawerTrigger>
  <DrawerContent>
    <DrawerHeader>
      <DrawerTitle>确定要继续吗？</DrawerTitle>
      <DrawerDescription>该操作无法撤销。</DrawerDescription>
    </DrawerHeader>
    <DrawerFooter>
      <Button>提交</Button>
      <DrawerClose>
        <Button variant="outline">取消</Button>
      </DrawerClose>
    </DrawerFooter>
  </DrawerContent>
</Drawer>
```
